//Clonar un branch en especifico

git clone -b nombre_de_la_rama link_del_repositorio
git clone -b nombre_de_la_rama --single-branch link_del_repositorio

// Creacion
git branch nombre_de_la_rama -> Crear nueva rama
git checkout nombre_de_la_rama -> Sirve para cambiar la rama actual
// O directamente
git checkout -b nombre_de_la_rama -> Creo nueva rama y me cambio a ella

// Comandos de branch

git branch -> Muestra todas las ramas del proyecto

git branch -m nombre_de_la_rama nuevo_nombre_rama -> Cambio nombre de una rama

git branch -d nuevo_nombre_rama -> Elimino una rama

git branch -h -> Muestra todos los usos posibles de git branch

// Realizando commits en la rama actual

git status -> Me fijo los cambios
git add .
git commit -m "Comentario"
git push -u origin nombre_de_la_rama

// Hacer pull de algun archivo en especifico

git checkout origin/nombre_de_la_rama nombre_del_archivo

// Realizar merge de rama completa

git checkout master
// Si hay algun error -> git fetch origin nombre_de_la_rama y despues hago merge
git merge nombre_de_la_rama

// Realizar merge a master de un archivo particular de una rama

git checkout master
git checkout nombre_de_la_rama nombre_del_archivo
